Table Students {
    student_id INT [pk, increment]
    first_name VARCHAR(100) [not null]
    last_name VARCHAR(100) [not null]
    email VARCHAR(150) [unique, not null]
    contact_number VARCHAR(15)
    date_of_birth DATE
    address VARCHAR(250)
    enrollment_date DATE
    status ENUM('active', 'inactive') [default: 'active']
}

Table Instructors {
    instructor_id INT [pk, increment]
    first_name VARCHAR(100) [not null]
    last_name VARCHAR(100) [not null]
    email VARCHAR(150) [unique, not null]
    contact_number VARCHAR(15)
}

Table Courses {
    course_id INT [pk, increment]
    course_name VARCHAR(150) [not null]
    start_date DATE
    instructor_id INT [ref: > Instructors.instructor_id]
}

Table Lessons {
    lesson_id INT [pk, increment]
    lesson_title VARCHAR(150) [not null]
    content TEXT
    course_id INT [ref: > Courses.course_id]
}

Table Assignments {
    assignment_id INT [pk, increment]
    title VARCHAR(150) [not null]
    description TEXT
    due_date DATE
    course_id INT [ref: > Courses.course_id]
}

Table Submissions {
    submission_id INT [pk, increment]
    assignment_id INT [ref: > Assignments.assignment_id]
    student_id INT [ref: > Students.student_id]
    submission_date DATE
    content TEXT
}

Table Batches {
    batch_id INT [pk, increment]
    batch_name VARCHAR(100) [not null]
    course_id INT [ref: > Courses.course_id]
    start_date DATE [not null]
    end_date DATE
}

Table Enrollments {
    enrollment_id INT [pk, increment]
    student_id INT [ref: > Students.student_id]
    batch_id INT [ref: > Batches.batch_id]
    enrollment_date DATE [not null]
    status ENUM('active', 'completed', 'dropped') [default: 'active']
}

Table Feedback {
    feedback_id INT [pk, increment]
    student_id INT [ref: > Students.student_id]
    course_id INT [ref: > Courses.course_id]
    instructor_id INT [ref: > Instructors.instructor_id]
    feedback_text TEXT
    rating INT [not null, note: "Rating from 1 to 5"]
    submission_date DATE [not null]
}

Table Attendance {
    attendance_id INT [pk, increment]
    student_id INT [ref: > Students.student_id]
    lesson_id INT [ref: > Lessons.lesson_id]
    date DATE [not null]
    status ENUM('present', 'absent', 'excused') [default: 'present']
}

Table Certificates {
    certificate_id INT [pk, increment]
    student_id INT [ref: > Students.student_id]
    course_id INT [ref: > Courses.course_id]
    issue_date DATE [not null]
    certificate_url VARCHAR(255)
}

Table Leaderboard {
    rank INT [pk, increment]
    student_id INT [ref: > Students.student_id]
    batch_id INT [ref: > Batches.batch_id]
    learning DECIMAL(5, 2) [not null]  // Assuming this represents a score or metric
}
